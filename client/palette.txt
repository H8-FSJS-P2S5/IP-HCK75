import { useEffect, useState } from "react";
import Card from "../component/Card";
import { axiosIns } from "../axios";

export default function Homepage() {
  const [posts, setPosts] = useState([]);
  const [disasterMessage, setDisasterMessage] = useState("");
  const [news, setNews] = useState([]);

  const getArticlesAndDisasterInfo = async () => {
    try {
      const { data } = await axiosIns.get("/article");
      setDisasterMessage(data.message); // Set the most common disaster message
      setNews(data.news); // Set news articles fetched from SerpApi
      setPosts(data.articles); // Set articles from the database
    } catch (error) {
      console.log("Error fetching data:", error);
    }
  };

  useEffect(() => {
    getArticlesAndDisasterInfo();
  }, []);

  return (
    <div className="bg-seashell min-h-screen">
      <div className="container mx-auto py-10">
        {/* Display the sentence about the most common disaster */}
        <h1 className="text-4xl font-bold text-center text-chamoisee mb-10">
          {disasterMessage}
        </h1>

        {/* Display recent disaster news fetched from SerpApi */}
        <div className="mb-10">
          <h2 className="text-2xl font-bold mb-4">Recent Natural Disaster News:</h2>
          <ul className="list-disc list-inside">
            {news.map((item, index) => (
              <li key={index}>
                <a href={item.link} target="_blank" rel="noopener noreferrer" className="text-light-blue hover:underline">
                  {item.title} - {item.source}
                </a>
              </li>
            ))}
          </ul>
        </div>

        {/* Articles grid */}
        <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-6">
          {posts.map((post) => (
            <Card key={post.id} post={post} />
          ))}
        </div>
      </div>
    </div>
  );
}
